13  package biz.gabrys.lesscss.extended.compiler.util;
14  
15  /**
16   * Provides tools to work with method parameters.
17   * @since 2.1.0
18   */
19  public final class ParameterUtils {
20  
21      private ParameterUtils() {
22          // blocks the possibility of create a new instance
23      }
24  
25      /**
26       * Verifies that a parameter value is not {@code null}.
27       * @param name the parameter name.
28       * @param value the parameter value.
29       * @throws IllegalArgumentException if the parameter value is equal to {@code null}.
30       * @since 2.1.0
31       */
32      public static void verifyNotNull(final String name, final Object value) {
33          if (value == null) {
34              final String message;
35              if (name == null) {
36                  message = "Parameter is null";
37              } else {
38                  message = String.format("Parameter \"%s\" is null", name);
39              }
40              throw new IllegalArgumentException(message);
41          }
42      }
43  }
